                results.add(batchResult);
            }
            return results;
        } finally {

            for (Statement stmt : statementList) {
                closeStatement(stmt);
            }
            // ç½®ç©º currentSqlã€statementListã€batchResultList å±æ€§
            currentSql = null;
            statementList.clear();

        }
    }
